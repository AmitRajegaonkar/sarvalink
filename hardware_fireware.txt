// lcd setup
LiquidCrystal_I2C lcd(0x27, 16, 2);

const char *ssid = "7i";             // Your WiFi SSID
const char *password = "allright0";  // Your WiFi Password

#define NTP_SERVER "in.pool.ntp.org"
#define UTC_OFFSET 19800              // 5 hours and 30 minutes in seconds
#define UTC_OFFSET_DST 0

#define RED_LED_PIN 32
#define GREEN_LED_PIN 1
#define BLUE_LED_PIN 16

// RFID setup
#define SS_PIN 5
#define RST_PIN 4
MFRC522 rfid(SS_PIN, RST_PIN);

// Keypad setup
const byte ROWS = 4;
const byte COLS = 3;
char keys[ROWS][COLS] = {
  { '1', '2', '3' },
  { '4', '5', '6' },
  { '7', '8', '9' },
  { '*', '0', '#' }
};
byte rowPins[ROWS] = { 13, 12, 14, 27 };
byte colPins[COLS] = { 26, 25, 33 };
Keypad keypad = Keypad(makeKeymap(keys), rowPins, colPins, ROWS, COLS);

#define BUZZER_PIN 2

void buzz(int duration) {
  digitalWrite(BUZZER_PIN, HIGH);  // Turn the buzzer on
  delay(duration);                 // Wait for the specified duration
  digitalWrite(BUZZER_PIN, LOW);   // Turn the buzzer off
}

// Variables to store data
String uidString = "";
String lastUID = "";
bool eventMode = false;
unsigned long lastReadTime = 0;
String enteredNumber = "";
String lastEnteredNumber = "";
unsigned long lastNumberTime = 0;
bool isRFIDMode = true;
TaskHandle_t rfidTaskHandle;
// Function Prototypes

void keypadTask(void *pvParameters);
void wifiTask(void *pvParameters);

// Function to get the subject based on the current time
String getCurrentSubject(int hour) {
  if (hour >= 1 && hour < 2) {
    return "OOPM";
  } else if (hour >= 2 && hour < 3) {
    return "DLCO";
  } else if (hour >= 23 && hour < 24) {
    return "CG";
  } else if (hour >= 22 && hour < 23) {
    return "DS";
  } else if (hour >= 11 && hour < 12) {
    return "M3";
  } else {
    return "";  // Default message if no class is scheduled
  }
}

// Function to print the local time on the lcd
void printLocalTime(bool isEventMode = false) {
  struct tm timeinfo;
  getLocalTime(&timeinfo);

  // Determine whether it's AM or PM
  bool isPM = timeinfo.tm_hour >= 12;

  // Convert hour to 12-hour format
  int hour12 = timeinfo.tm_hour % 12;
  if (hour12 == 0) {
    hour12 = 12;  // Handle midnight case
  }

  // If in event mode, show "EM-ON"
  String subject = isEventMode ? "    \u1F512" : getCurrentSubject(timeinfo.tm_hour);

  // Print the time and subject on the LCD
  lcd.setCursor(0, 0);
  lcd.printf("%02d:%02d %s", hour12, timeinfo.tm_min, isPM ? "PM" : "AM");
  lcd.setCursor(10, 0);
  lcd.printf("%s", subject.c_str());
}



void setup() {
  Serial.begin(115200);

  pinMode(BUZZER_PIN, OUTPUT);

  // Initialize the LED pins as outputs
  pinMode(RED_LED_PIN, OUTPUT);
  pinMode(GREEN_LED_PIN, OUTPUT);
  pinMode(BLUE_LED_PIN, OUTPUT);
  digitalWrite(RED_LED_PIN, LOW);
  digitalWrite(GREEN_LED_PIN, LOW);
  digitalWrite(BLUE_LED_PIN, HIGH);

  // lcd initialization
  lcd.init();
  lcd.backlight();

  // RFID initialization
  SPI.begin();
  rfid.PCD_Init();

  lcd.clear();
  printLocalTime();  
  lcd.setCursor(0, 1);
  lcd.print("TAP CARD");

  // Create FreeRTOS tasks

  xTaskCreate(keypadTask, "Keypad Task", 4096, NULL, 1, NULL);

  xTaskCreate(rfidTask, "RFID Task", 2048, NULL, 1, &rfidTaskHandle);

  xTaskCreate(wifiTask, "WiFi Task", 2048, NULL, 1, NULL);


}

void loop() {
  // FreeRTOS tasks handle everything, so no need to do anything in loop().
}

void configureTime() {
  // Configure NTP with the specified server and offsets
  configTime(UTC_OFFSET, UTC_OFFSET_DST, NTP_SERVER);
}

// Task to handle RFID operations
void rfidTask(void *pvParameters) {
  while (true) {
    if (isRFIDMode) {
      // Check if an RFID card is present
      if (rfid.PICC_IsNewCardPresent() && rfid.PICC_ReadCardSerial()) {
        uidString = "";
        for (byte i = 0; i < rfid.uid.size; i++) {
          uidString += String(rfid.uid.uidByte[i], HEX);
        }
        uidString.toUpperCase();

        unsigned long currentTime = millis();

        // Check if the same card is detected within 2 seconds
        if (uidString == lastUID && (currentTime - lastReadTime <= 2000)) {
          lcd.clear();
          lcd.setCursor(0, 0);
          lcd.print("Same Card");
          buzz(500);
          Serial.println("Same Card: " + uidString);
        } else {
          lcd.clear();
          lcd.setCursor(0, 0);
          lcd.print("SUCCESS");
          buzz(200);
          Serial.println("New Card: " + uidString);
          lastUID = uidString;         // Update lastUID
          lastReadTime = currentTime;  // Update lastReadTime
          sendUIDToServer(uidString);
          sendUIDToServer_admin(uidString);
        }

        vTaskDelay(pdMS_TO_TICKS(200));  // Wait 2 seconds before clearing the screen
        lcd.clear();
        printLocalTime();  
        lcd.setCursor(0, 1);
        lcd.print("TAP CARD");
      }
    }

    vTaskDelay(pdMS_TO_TICKS(100));  // Yield to other tasks
  }
}

TaskHandle_t sendTaskHandle = NULL;

void asyncSendAttendanceToServer(void* parameters) {
  String* args = (String*)parameters; // Pass all parameters as strings
  
  HTTPClient http;
  http.begin(args[0]);  // The URL
  http.addHeader("Content-Type", "application/x-www-form-urlencoded");

  int httpResponseCode = http.POST(args[1]);  // Post data

  if (httpResponseCode > 0) {
    String response = http.getString();
    Serial.println("Server Response: " + response);
  } else {
    Serial.println("Error on sending POST: " + String(httpResponseCode));
  }

  http.end();  // Close the connection
  vTaskDelete(NULL);  // Kill the task after completion
}

void sendAttendanceToServer(String uid, String key_uid, String event = "", bool isAdmin = false) {
  if (WiFi.status() == WL_CONNECTED) {
    String url = isAdmin ? "http://172.25.21.190:1254/uid" : "http://172.25.21.190:1264/store_attendance";

    // Prepare POST data
    String postData = "";
    if (uid != "") postData += "uid=" + uid;
    if (key_uid != "") {
      if (postData != "") postData += "&";
      postData += "key_uid=" + key_uid;
    }
    if (event != "") postData += "&event=" + event;

    // Run in a separate FreeRTOS task to avoid blocking the main thread
    String* args = new String[2]{url, postData};
    xTaskCreate(asyncSendAttendanceToServer, "sendAttendanceTask", 8192, (void*)args, 1, &sendTaskHandle);
  } else {
    Serial.println("WiFi Disconnected. Cannot send data to server.");
  }
}



// Function to send UID to the server

// For sending only RFID UID to the server
void sendUIDToServer_admin(String uid) {
  sendAttendanceToServer(uid, "", "", true);
}

// For sending only keypad key UID to the server
void sendKeyToServer_admin(String key_uid) {
  sendAttendanceToServer("", key_uid, "", true);
}

// For sending only RFID UID to the server
void sendUIDToServer(String uid) {
  sendAttendanceToServer(uid, "", "", false);  // Send only UID
}

// For sending only keypad key UID to the server
void sendKeyToServer(String key_uid) {
  sendAttendanceToServer("", key_uid, "", false);  // Send only key_uid
}

// For sending both UID and Key UID to the server with an event
void sendEventrfidToServer(String uid, String event) {
  sendAttendanceToServer(uid, "", event, false);  // Send both UID, Key UID, and event
}

// For sending event no with Key to the server with an event
void sendEventKeyToServer(String key_uid, String event) {
  sendAttendanceToServer("", key_uid, event, false);  // Send both UID, Key UID, and event
}


// Task to handle Keypad operations
void keypadTask(void *pvParameters) {
  const unsigned long debounceDelay = 50;  // 50ms debounce time
  unsigned long lastDebounceTime = 0;
  char lastKey = '\0';
  bool eventMode = false;
  String eventNumber = "";
  String uidNumber = "";
  String enteredNumber = "";
  String lastEnteredNumber = "";
  unsigned long lastNumberTime = 0;

  while (true) {
    char key = keypad.getKey();  // Check if a key is pressed
    if (key) {
      // Debounce logic
      if ((millis() - lastDebounceTime) > debounceDelay) {
        lastDebounceTime = millis();
        if (key == '#' && lastKey == '*') {

          eventMode = !eventMode;  // Turn off event mode
          if (eventMode == false) {
            vTaskResume(rfidTaskHandle);
            printLocalTime(eventMode);
            lcd.clear();      
            lcd.setCursor(0, 1);
            lcd.print("UID:");
            lcd.print(enteredNumber);

          } else {
            vTaskSuspend(rfidTaskHandle);
            lcd.clear();
            printLocalTime(eventMode);
            lcd.setCursor(0, 1);
            lcd.print("Event No:");
          }



        } else if (eventMode) {

          if (key >= '0' && key <= '9' && eventNumber.length() < 6) {
            eventNumber += key;
            
            lcd.setCursor(0, 1);
            lcd.print("Event No:" + eventNumber);
          } else if (key == '*' && eventNumber.length() > 0) {
            // Handle backspace for event number
            eventNumber.remove(eventNumber.length() - 1);
            lcd.clear();
            printLocalTime(eventMode);
            lcd.setCursor(0, 1);
            lcd.print("Event No:" + eventNumber);
          } else if (key == '#' && eventNumber.length() == 6) {
            // Event number completed, proceed to UID input

            lcd.clear();
            printLocalTime(eventMode);
            lcd.setCursor(0, 1);
            lcd.print("UID:");
            uidNumber = "";

            // Handle UID input
            while (true) {
              key = keypad.getKey();
              if (rfid.PICC_IsNewCardPresent() && rfid.PICC_ReadCardSerial()) {
                uidString = "";
                for (byte i = 0; i < rfid.uid.size; i++) {
                  uidString += String(rfid.uid.uidByte[i], HEX);
                }
                uidString.toUpperCase();

                unsigned long currentTime = millis();

                lcd.clear();
                lcd.setCursor(0, 0);
                lcd.print("SUCCESS");
                sendEventrfidToServer(uidString, eventNumber);
                buzz(200);
                Serial.println("New Card: " + uidString);
                lastUID = uidString;         // Update lastUID
                lastReadTime = currentTime;  // Update lastReadTime
                delay(200);                  // Wait 2 seconds before clearing the screen
                lcd.clear();
                eventNumber = "";
                uidNumber = "";
                printLocalTime(eventMode);
                lcd.setCursor(0, 1);
                lcd.print("Event No:" + eventNumber);
                break;
              } else if (key >= '0' && key <= '9' && uidNumber.length() < 11) {
                uidNumber += key;
                lcd.setCursor(0, 1);
                lcd.print("UID:" + uidNumber);
              } else if (key == '*' && uidNumber.length() > 0) {
                // Handle backspace for UID number
                uidNumber.remove(uidNumber.length() - 1);
                lcd.clear();
                printLocalTime(eventMode);
                lcd.setCursor(0, 1);
                lcd.print("UID:" + uidNumber);
              } else if (key == '#' && uidNumber.length() == 11) {
                // UID number completed
                lcd.clear();
                lcd.setCursor(0, 0);
                lcd.print("SUCCESS ");
                delay(100);
                lcd.clear();
                printLocalTime(eventMode);
                lcd.setCursor(0, 1);
                lcd.print("Event No:");
                sendEventKeyToServer(uidNumber, eventNumber);
                eventNumber = "";
                uidNumber = "";
                break;
              }

              vTaskDelay(pdMS_TO_TICKS(10));  // Small delay to allow other tasks to run
            }
          }

        } else if (!eventMode) {

          if (key >= '0' && key <= '9' && enteredNumber.length() < 11) {

            enteredNumber += key;
            lcd.clear();
            printLocalTime();
            lcd.setCursor(0, 1);
            lcd.print("UID:");
            lcd.print(enteredNumber);  // Display number on lcd
          } else if (key == '*' && enteredNumber.length() > 0) {
            enteredNumber.remove(enteredNumber.length() - 1);  // Backspace
            lcd.clear();
            printLocalTime();
            lcd.setCursor(0, 1);
            lcd.print("UID:");
            lcd.print(enteredNumber);
          } else if (key == '#' && enteredNumber.length() == 11) {

            unsigned long currentTime = millis();

            // Check if the same number is entered within 10 seconds
            if (enteredNumber == lastEnteredNumber && (currentTime - lastNumberTime <= 5000)) {
            lcd.clear();
            printLocalTime();
            lcd.setCursor(0, 1);
            lcd.print("UID:");
            lcd.print(enteredNumber);
              buzz(500);
              Serial.println("Same Number: " + enteredNumber);
            } else {
              lcd.clear();
              printLocalTime();
              lcd.setCursor(0, 1);
              lcd.print("SUCCESS");
              buzz(200);

              Serial.println("New Number: " + enteredNumber);
              lastEnteredNumber = enteredNumber;  // Update lastEnteredNumber
              lastNumberTime = currentTime;       // Update lastNumberTime
            }

            vTaskDelay(pdMS_TO_TICKS(200)); // Optional: Small delay to make sure the user sees the message
            lcd.clear();
            printLocalTime();
            lcd.setCursor(0, 1);
            lcd.print("UID:");
            sendKeyToServer(enteredNumber);
            sendKeyToServer_admin(enteredNumber);
            enteredNumber = "";  // Reset the entered number
          }
        }

        lastKey = key;  // Store the last key press
      }
    }

    vTaskDelay(pdMS_TO_TICKS(10));  // Small delay to allow other tasks to run
  }
}


// Task to handle WiFi connection
void wifiTask(void *pvParameters) {
  while (true) {
    if (WiFi.status() != WL_CONNECTED) {
      Serial.print("Connecting to WiFi");
      WiFi.begin(ssid, password);
      while (WiFi.status() != WL_CONNECTED) {
        digitalWrite(RED_LED_PIN, HIGH);
        digitalWrite(GREEN_LED_PIN, LOW);
        vTaskDelay(pdMS_TO_TICKS(500));  // Wait for 500ms before retrying
      }
      Serial.println("\nConnected to WiFi!");
      Serial.print("IP address: ");
      Serial.println(WiFi.localIP());
      
      // Configure NTP time
      configureTime();
      delay(1000);  // Give some time for the NTP client to sync
    } else {
      // WiFi is connected
      digitalWrite(RED_LED_PIN, LOW);
      digitalWrite(GREEN_LED_PIN, HIGH);

      // Print local time only if the time is successfully retrieved
      struct tm timeinfo;
      if (getLocalTime(&timeinfo)) {
        printLocalTime();  // Call this only if time retrieval is successful
      } else {
        Serial.println("Failed to obtain time");
      }

      delay(100);  // Delay to allow other processes
    }

    vTaskDelay(pdMS_TO_TICKS(1000));  // Check WiFi status every second
  }
}
